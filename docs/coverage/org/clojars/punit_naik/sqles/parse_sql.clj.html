001  (ns org.clojars.punit-naik.sqles.parse-sql
002    (:require [clojure.string :as str]
003              [org.clojars.punit-naik.sqles.query :as query]
004              [org.clojars.punit-naik.sqles.parse-sql.utils :refer [handle-clause-data]]
005              [org.clojars.punit-naik.sqles.parse-sql.where :as where]))
006  
007  (defn query->es-op
008    [clause]
009    (let [clause (str/lower-case clause)]
010      (get {"select" "/_search"}
011           clause)))
012  
013  (defn clause->query-fn
014    [clause]
015    (when clause
016      (let [clause (str/lower-case clause)]
017        (get {"select" query/select
018              "where" query/where->es
019              "from" (constantly nil)}
020             clause))))
021  
022  (defn find-index
023    "Finds the index to be queried from the SQL query"
024    [sql-query]
025    (let [parts (->> (str/split sql-query #" ")
026                     (map-indexed vector)
027                     (into {}))
028          from-clauses #{"from" "FROM"}]
029      (->> (filter (fn [[k _]]
030                     (contains? from-clauses (get parts (dec k)))) parts)
031           vals first)))
032  
033  (defn take-till-next-clause
034    "Gets the data for a particular clause from query parts
035     Till the next clause is found"
036    [query-parts]
037    (take-while #(not (clause->query-fn %)) query-parts))
038  
039  (defmethod handle-clause-data "select"
040    [_ clause-data]
041    (map str/trim
042         (mapcat (fn [cd]
043                   (remove empty? (str/split cd #",")))
044                 clause-data)))
045  
046  (defmethod handle-clause-data "where"
047    [_ clause-data]
048    (where/handle-clause-data clause-data))
049  
050  (defn clean-query
051    "Removes spaces before and after commas
052     And spaces after opened round bracket and before closed round bracket
053     Won't match commas/round brackets between quotes, single or double"
054    [query]
055    (-> (str/replace query #"(?!\B\"[^(\"|\'|\`)]*)[\s+]?,(?![^(\"|\'|\`)]*(\"|\'|\`)\B)\s+" ",")
056        (str/replace #"\((?![^(\"|\'|\`)]*(\"|\'|\`)\B)\s+" "(")
057        (str/replace #"(?!\B\"[^(\"|\'|\`)]*)[\s+]?\)" ")")
058        (str/replace #"NULL|null" "nil")))
059  
060  (defn parse-query
061    [sql-query]
062    (let [sql-query (clean-query sql-query)
063          index (find-index sql-query)
064          parts (str/split sql-query
065                           #"\s+(?=([^\"\'\`]*[\"|\'|\`][^\"\'\`]*[\"|\'|\`])*[^\"\'\`]*$)")]
066      (loop [ps parts
067             result {:url (str (query/from index)
068                               (query->es-op (first parts)))
069                     :body {}}]
070        (if (empty? ps)
071          result
072          (let [clause (str/lower-case (first ps))
073                clause-data (take-till-next-clause (rest ps))
074                remaining-parts (drop (count clause-data) (rest ps))
075                clause-data (handle-clause-data clause clause-data)
076                intermediate-es-query ((clause->query-fn clause) clause-data)]
077            (recur remaining-parts
078                   (update result :body merge intermediate-es-query)))))))
