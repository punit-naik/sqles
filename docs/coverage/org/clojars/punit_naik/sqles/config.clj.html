001  (ns org.clojars.punit-naik.sqles.config
002    (:require [omniconf.core :as cfg]
003              [clojure.string :as str]
004              [clj-http.client :as http]
005              [cheshire.core :as json]))
006  
007  (defmacro when-let-multiple
008    [bindings & body]
009    `(let ~bindings
010       (if (and ~@(take-nth 2 bindings))
011         (do ~@body))))
012  
013  (cfg/define
014    {:es-protocol {:description "Protocol to contact the ES Server"
015                   :type :string
016                   :one-of ["http" "https"]
017                   :default "http"}
018     :es-hostname {:description "Where service is deployed"
019                   :type :string
020                   :required true}
021     :es-port {:description "HTTP port"
022               :type :number
023               :required true}
024     :es-username {:description "Username for the ES Server"
025                   :type :string}
026     :es-password {:description "Password for the ES Server"
027                   :type :string
028                   :secret true}
029     :conf {:description "Configuration file"
030            :type :file
031            :verifier cfg/verify-file-exists}})
032  
033  (cfg/populate-from-properties)
034  (when-let [conf (cfg/get :conf)]
035    (cfg/populate-from-file conf))
036  (cfg/populate-from-env)
037  (cfg/verify :quit-on-error true)
038  
039  (defn add-trailing-backslash
040    [link]
041    (if-not (str/ends-with? link "/")
042      (str link "/") link))
043  
044  (defn ping-server
045    [server]
046    (try (let [server-url (str (add-trailing-backslash server) "_cluster/health")
047               {:keys [body status]} (http/get server-url {:accept :json})]
048           {:body (json/parse-string body true)
049            :status status})
050         (catch Exception _ {:status 405})))
051  
052  (defn server-up?
053    [server]
054    (let [{:keys [status] {:keys [number_of_nodes]} :body} (ping-server server)]
055      (and (= status 200)
056           (>= number_of_nodes 1))))
057  
058  (defn generate-server-url-from-config
059    []
060    (str (cfg/get :es-protocol) "://"
061         (when-let-multiple
062          [username (cfg/get :es-username)
063           password (cfg/get :es-password)]
064          (str username ":" password "@"))
065         (cfg/get :es-hostname)
066         (when-let [port (cfg/get :es-port)]
067           (str ":" port))))
068  
069  (defn es-server
070    []
071    (let [url (generate-server-url-from-config)
072          is-server-up? (server-up? url)]
073      (when is-server-up?
074        (add-trailing-backslash url))))
